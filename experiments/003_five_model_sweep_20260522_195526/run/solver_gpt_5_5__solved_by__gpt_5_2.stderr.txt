            mm=re.match(r\"the request class belongs to the defined group (\\w+)"'$",p)
            if mm:
                cond['"'group']=mm.group(1); continue
            mm=re.match(r\"the file carries the (\\w+) marker"'$",p)
            if mm:
                cond.setdefault('"'tokens',set()).add(mm.group(1)); continue
            mm=re.match(r\"the file carries the bulk marker"'$",p)
            if mm:
                cond.setdefault('"'tokens',set()).add('bulk'); continue
            raise ValueError(f\"unknown condition fragment: {p} in {text}\")
    return rid, Rule(rid,status=='active',prio,cond,fee,deadline,notice_required)


def parse_base(base_code):
    base=ensure_obj(base_code)
    defs={}
    for d in base['definitions']:
        m=re.match(r\"(\\w+) initially contains (.+)\\.\",d.strip())
        defs[m.group(1)]=set(x.strip() for x in m.group(2).split(' and '))
    caps={}
    for c in base['caps']:
        m=re.match(r\"Initial cap: region=(\\w+), request_class=(\\w+), fee_cap=(\\d+)\\.\",c.strip())
        caps[(m.group(1),m.group(2))]=int(m.group(3))
    rules={}
    for r in base['rules']:
        rid,rule=parse_rule_text(r)
        rules[rid]=rule
    return defs,caps,rules


def apply_amendments(defs,caps,rules,waivers,amendments):
    for a in amendments:
        a=a.strip()
        m=re.match(r\"A\\d+: Suspend (R\\d+)\\.\",a)
        if m:
            rid=m.group(1)
            if rid in rules:
                rules[rid].active=False
            continue
        m=re.match(r\"A\\d+: Replace (R\\d+) in full with: (R\\d+ \\(.*)"'$",a)
        if m:
            rid=m.group(1)
            _,nr=parse_rule_text(m.group(2))
            rules[rid]=nr
            continue
        m=re.match(r"A'"\\d+: Add rule (R\\d+): (R\\d+ \\(.*)"'$",a)
        if m:
            rid=m.group(1)
            _,nr=parse_rule_text(m.group(2))
            rules[rid]=nr
            continue
        m=re.match(r"A'"\\d+: For region=(\\w+) and request_class=(\\w+), set the fee cap to (\\d+) fee_units\\.\",a)
        if m:
            caps[(m.group(1),m.group(2))]=int(m.group(3))
            continue
        m=re.match(r\"A\\d+: Definition update: (include|exclude) (\\w+) (?:in|from) (\\w+)\\.\",a)
        if m:
            op,item,group=m.group(1),m.group(2),m.group(3)
            defs.setdefault(group,set())
            (defs[group].add(item) if op=='include' else defs[group].discard(item))
            continue
        m=re.match(r\"A\\d+: Add notice waiver (W\\d+): notice is waived when any one of these tokens is present in the case file: (.+)\\.\",a)
        if m:
            waivers[m.group(1)]=set(t.strip() for t in m.group(2).split(','))
            continue
        raise ValueError(f\"unparsed amendment: {a}\")


def case_tokens(case):
    toks=set()
    if isinstance(case,dict):
        for v in case.values():
            if isinstance(v,str): toks.add(v)
            elif isinstance(v,list): toks.update(x for x in v if isinstance(x,str))
            elif isinstance(v,dict): toks|=case_tokens(v)
    return toks


def rule_matches(rule,case,defs):
    if not rule.active: return False
    if rule.cond.get('any'): return True
    req=case.get('kind') or case.get('request_class')
    if 'request_class' in rule.cond and rule.cond['request_class']"'!=req: return False
    if '"'region' in rule.cond and rule.cond['region']"'!=case.get('"'region'): return False
    if 'channel' in rule.cond and rule.cond['channel']"'!=case.get('"'channel'): return False
    if 'group' in rule.cond and req not in defs.get(rule.cond['group'],set()): return False
    if 'tokens' in rule.cond:
        toks=case_tokens(case)
        if not (toks & rule.cond['tokens']):
            return False
    return True


def solve(item):
    defs,caps,rules=parse_base(item['base_code'])
    case=ensure_obj(item['case_file'])
    waivers={}
    apply_amendments(defs,caps,rules,waivers,item['amendments'])
    matches=[r for r in rules.values() if rule_matches(r,case,defs)]
    matches.sort(key=lambda r:(-r.priority,r.id))
    win=matches[0]
    req=case.get('kind') or case.get('request_class')
    fee=min(win.fee_units,caps[(case['region'],req)])
    toks=case_tokens(case)
    mw=sorted(w for w,ts in waivers.items() if toks & ts)
    notice='no' if (not win.notice_required) or mw else 'yes'
    waiver='+'.join(mw) if mw else 'none'
    return f\"rule={win.id}|fee_units={fee}|deadline_days={win.deadline_days}|notice={notice}|waiver={waiver}\"

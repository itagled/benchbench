    # Examples:
    # If zone is "'`West` and category is `Agriculture`, replace the entire risk score with `1`.
    m=re.match(r"If (.+), replace the entire risk score with `('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            return True, int(m.group(2)), fee
        return False, risk, fee
    m=re.match(r\"If (.+), replace the running fee with "'`('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            return True, risk, int(m.group(2))
        return False, risk, fee
    # min then add fee
    m=re.match(r\"If (.+), replace the risk score with the smaller of its current value and "'`('"\\d+)"'`; then add fee `('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            risk=min(risk, int(m.group(2)))
            fee+=int(m.group(3))
            return True, risk, fee
        return False, risk, fee
    raise ValueError(f\"Unparsed replacement rule: {rule}\")


def apply_additive_rule(rule:str, case:dict, risk:int, fee:int):
    # returns (risk, fee)
    m=re.match(r\"If (.+), add fee "'`('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            fee+=int(m.group(2))
        return risk, fee
    m=re.match(r\"If (.+), add risk "'`('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            risk+=int(m.group(2))
        return risk, fee
    m=re.match(r\"If (.+), subtract fee "'`('"\\d+)"'` if (.+)'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee) and eval_condition(m.group(3), case, risk, fee):
            fee-=int(m.group(2))
        return risk, fee
    m=re.match(r\"If (.+), subtract fee "'`('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            fee-=int(m.group(2))
        return risk, fee
    raise ValueError(f\"Unparsed additive rule: {rule}\")


def waiver_reduction(rule:str, case:dict, risk:int, fee:int):
    # return reduction amount (positive) if applies else 0
    m=re.match(r\"If (.+), subtract fee "'`('"\\d+)"'`'"\\.?\", rule)
    if m:
        if eval_condition(m.group(1), case, risk, fee):
            return int(m.group(2))
        return 0
    raise ValueError(f\"Unparsed waiver: {rule}\")


def apply_final_review(rule:str, case:dict, risk:int, fee:int):
    # only known pattern: force risk score at least N based on manual_note and fee
    m=re.match(r\"If (.+) and (.+), force the final risk score to be at least "'`('"\\d+)"'`'"\\.?\", rule)
    if m:
        # second condition may reference final fee before lane assignment; pass fee
        if eval_condition(m.group(1), case, risk, fee, final_fee_before_lane=fee) and eval_condition(m.group(2), case, risk, fee, final_fee_before_lane=fee):
            risk=max(risk, int(m.group(3)))
        return risk
    raise ValueError(f\"Unparsed final review: {rule}\")


def adjudicate(case, charter, amendments):
    risk=charter['zone_risk'][case['zone']] + charter['tier_risk'][case['tier']] + charter['cat_risk'][case['category']]
    fee=charter['base_fee_start'] + case['quantity']*charter['base_fee_per_qty']
    risk, fee = apply_penalties(charter['penalties'], case, risk, fee)

    # replacements in order listed
    for r in amendments['Replacement rules']:
        applied, risk, fee = apply_replacement_rule(r, case, risk, fee)
    # additives
    for r in amendments['Additive rules']:
        risk, fee = apply_additive_rule(r, case, risk, fee)
    # waivers: choose max reduction
    best=0
    for r in amendments['Waivers']:
        best=max(best, waiver_reduction(r, case, risk, fee))
    fee -= best
    if fee<0: fee=0
    # final review trigger
    for r in amendments['Final review trigger']:
        risk = apply_final_review(r, case, risk, fee)

    # verdict
    if risk<=charter['allow_max']:
        verdict='ALLOW'
    elif risk<=charter['review_max']:
        verdict='REVIEW'
    else:
        verdict='DENY'
    # lane
    if fee<=charter['green_max']:
        lane='GREEN'
    elif fee<=charter['amber_max']:
        lane='AMBER'
    else:
        lane='RED'
    return f\"verdict={verdict};lane={lane};fee={fee}\"

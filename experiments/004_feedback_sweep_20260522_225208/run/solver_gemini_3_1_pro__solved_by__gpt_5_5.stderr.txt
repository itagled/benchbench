def crc16(data: bytes) -> bytes:
    """Computes a Modbus CRC16 over the given bytes."""
    crc = 0xFFFF
    for byte in data:
        crc ^= byte
        for _ in range(8):
            if crc & 1:
                crc = (crc >> 1) ^ 0xA001
            else:
                crc >>= 1
    return bytes([crc >> 8, crc & 0xFF])

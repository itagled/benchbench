from collections import deque

def min_steps(initial, target, rules, max_len=12, max_steps=25):
    if initial == target:
        return 0
    visited = {initial}
    frontier = [initial]
    for step in range(1, max_steps + 1):
        nxt = []
        for s in frontier:
            for pat, rep in rules:
                i = 0
                while True:
                    j = s.find(pat, i)
                    if j < 0: break
                    new_s = s[:j] + rep + s[j+2:]
                    if len(new_s) > max_len:
                        i = j + 1; continue
                    if new_s == target:
                        return step
                    if new_s not in visited:
                        visited.add(new_s)
                        nxt.append(new_s)
                    i = j + 1
        if not nxt:
            return -1
        frontier = nxt
    return -1
